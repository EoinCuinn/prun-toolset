function rebuildHypo() {
  const qty = id => { const el = document.getElementById(id); return el ? Math.max(0, +el.value || 0) : 0; };
  const val = id => document.getElementById(id)?.value || '';
  const C = COMPONENT_DATA;

  const hull = C.hull[val('hullSel')];
  const eng  = C.engine[val('engSel')];
  const engQty = qty('engSelQty');
  if (!hull || !eng) { _currentShip = null; document.getElementById('hypoMeta').innerHTML = 'Select at least a hull and engine.'; updateFuelSlider(); onLoadoutChange(); return; }

  let mass = hull.mass;
  mass += eng.mass * engQty;
  const add = (cat, sel, q) => { const c = C[cat][val(sel)]; if (c) mass += c.mass * q; };
  add('stlTank',   'stlTkSel',  qty('stlTkSelQty'));
  add('ftlTank',   'ftlTkSel',  qty('ftlTkSelQty'));
  add('cargo',     'cargoSel',  qty('cargoSelQty'));
  const seat = C.seat[val('seatSel')];
  if (seat) mass += seat.mass;
  document.querySelectorAll('.dmg-cb:checked').forEach(cb => { mass += C.damage[cb.value].mass; });

  const reactor = C.ftlReactor[val('reactorSel')];
  const cargoC = C.cargo[val('cargoSel')];
  const cargoCap = cargoC ? cargoC.capT * qty('cargoSelQty') : 0;

  // Shielding: general = hull plate + drones (additive); heat/whipple/radiation/gravity
  // from the checked damage components (spec §3a).
  const sh = { general: SHIELDING_FACTORS[val('hullSel')]?.general || 0, heat: 0, whipple: 0, radiation: 0, gravity: 0 };
  document.querySelectorAll('.dmg-cb:checked').forEach(cb => {
    const f = SHIELDING_FACTORS[cb.value] || {};
    for (const k of ['general', 'heat', 'whipple', 'radiation', 'gravity']) if (f[k]) sh[k] += f[k];
  });

  _currentShip = {
    source: 'hypo',
    label: `${val('hullSel')}+${engQty}×${val('engSel')}`,
    blueprintId: 'HYPOTHETICAL',
    condition: 1,
    engineType: eng.name,             // e.g. 'standardEngine' → drives landingDistKm() per-engine ratio
    thrust: eng.thrust * engQty,
    operatingEmptyMass: mass,
    maxGFactor: hull.maxGFactor + (seat ? seat.gBonus : 0),
    reactorPower: reactor ? REACTOR_POWER[val('reactorSel')] : null,
    chargeFactor: null,
    stlFlowRate: eng.flow * engQty,   // real per selected engine × count — drives TO/LND fuel + Fix 1 guard
    cargoCap,
    shielding: sh,
  };
  // Bound the cargo slider by the selected bay capacity (tonnes ceiling).
  setSliderMax('cargoMassSlider', Math.max(1, cargoCap || 5000));
  const cur = +document.getElementById('cargoMassSlider').value;
  if (cargoCap && cur > cargoCap) setSlider('cargoMassSlider', 'cargoMassVal', cargoCap, 't', 0);

  renderHypoMeta();
  updateFuelSlider();
  onLoadoutChange();
}

const STRUCT_COUNT = {
  plate: v => Math.round(0.535 * Math.pow(v, 0.655)), // hull plates of the chosen type
  ssc:   v => Math.round(0.0478 * v),                 // structuralSpacecraftComponent (weight 1 each)
};

HOW WEIGHTS ARE STORED/USED TODAY:
  * COMPONENT_DATA[cat][ticker].mass = FITTED AGGREGATE mass. For hull & cargo it
    bundles volume-scaling structural mass into one number (hull.RHP.mass=867.8,
    cargo.LCB.mass=1456.2 — the latter is literally a whole-ship empty mass).
  * COMPONENT_DATA[...].bomWeight = intended true per-unit BOM weight (partial coverage).
  * rebuildHypo() sums `.mass` (NOT bomWeight): mass = hull.mass + Σ eng/tank/cargo/seat/damage .mass.
    Reactor mass is NOT added at all. Structural parts (plates count, SSC, bridge, FFC,
    emitters, quarters) are NOT enumerated — they are implicitly inside hull.mass/cargo.mass.
  * => double-counting: hull.mass already bundles structural, and cargo.mass (esp LCB) also
    bundles near-whole-ship mass -> Hypothetical empty mass runs high (~19%+).

OTHER CODE DEPENDING ON operatingEmptyMass IN HYPO MODE:
  * renderHypoMeta() [1390] — displays s.operatingEmptyMass.
  * onLoadoutChange()/totalMass path [~1416+] — totalMass = empty + stl + ftl + cargo,
    feeds acceleration = thrust/totalMass and all fuel/landing calcs. So empty-mass error
    propagates into accel, TO/LND fuel, and flight time in Hypothetical mode.
  * _currentShip.operatingEmptyMass set at line 1372.

IS STEP 3 ALONE SUFFICIENT?  NO — explicitly confirmed.
  The old rebuildHypo summed FITTED `.mass` values and did NOT enumerate structural parts
  (hull-plate count, SSC, bridge, FFC, emitters, quarters); those were implicitly bundled
  into hull.mass + cargo.mass. Just correcting weights would drop all structural mass and
  make empty mass ~80% too LOW. The exact Σ(bomWeight×count) formula requires the FULL
  bill of material, so rebuildHypo had to be rewritten to RECONSTRUCT it, not just reweight.
